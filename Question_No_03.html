// Question 3: // Write a JavaScript program to create menu driven calculator that performs basic arithmetic operations (add, subtract, multiply and divide) using switch case. The calculator should input two numbers and an operator: +, -, *, / from the user. It should perform operation according to the operator entered and must take input in given format. let res; const operators = prompt("Enter operator ( either +, -, * or / ): "); const numb1 = parseFloat(prompt("Enter first number: ")); const numb2 = parseFloat(prompt("Enter second number: ")); switch (operators) { case "+": res = numb1 + numb2; console.log(`${numb1} + ${numb2} = ${res}`); break; case "-": res = numb1 - numb2; console.log(`${numb1} - ${numb2} = ${res}`); break; case "*": res = numb1 * numb2; console.log(`${numb1} * ${numb2} = ${res}`); break; case "/": res = numb1 / numb2; console.log(`${numb1} / ${numb2} = ${res}`); break; default: console.log("Invalid operator"); }
